\usepackage[a4paper,margin=1.5cm]{geometry}
%\usepackage[letterpaper,margin=1.5cm]{geometry}
\usepackage[T1]{fontenc}
\usepackage{tabularx}
\usepackage{multirow}
\usepackage{hyperref}
\urlstyle{same}
\setlength\parindent{0pt}

\newcolumntype{L}[1]{@{\hspace*{0.5cm}}
                     >{\raggedright\arraybackslash}p{#1}}
\newcolumntype{C}[1]{>{\centering  \arraybackslash}p{#1}}
\newcolumntype{R}[1]{>{\raggedleft \arraybackslash}p{#1}
                     @{\hspace*{0.1cm}}}
\arrayrulewidth=1.2pt
\pagenumbering{gobble}

\renewcommand{\title}[3]{
\begin{center}
\huge{\textbf{\textsc{#1}}}
\\[0.1in]
\end{center}

\begin{minipage}[t]{0.35\textwidth}
#2
\end{minipage}
\hfill
\begin{minipage}[t]{0.35\textwidth}
\flushright{%
%\vfill
#3
}%
\end{minipage}
\\[0.1in]
}

% customise the \section command 
\renewcommand{\section}[1]{
\vspace*{1ex}
\begin{tabularx}{18cm}{L{13cm}R{4cm}}
\hspace*{-0.5cm}\Large{\textbf{\textsc{#1}}} & \\
\hline
\end{tabularx}\\[-1pt] }

\renewcommand{\subsection}[3]{
\begin{tabularx}{18cm}{L{13cm}R{4cm}}
{\bf #1}\ifx#2\empty\else, #2\fi & #3
\end{tabularx}\\ }

\newcommand{\content}[1]{
\begin{tabularx}{18cm}{@{\hspace*{0.5cm}}p{17.5cm}}
#1
\end{tabularx}}

\newcommand{\bolditem}[2]{{\bf #1} #2 \\ }
\newcommand{\position}[1]{{\it #1}\\ }
\newcommand{\bulletitem}[1]{$\bullet$\hspace*{0.2cm} #1\\ }
\newcommand{\paper}[3]{#1, ``#2,''
\ifnum\pdfmatch {submi}{#3}=1
%submitted for review%
#3%
\else
\ifnum\pdfmatch{accepted}{#3}=1
#3%
\else%
in #3%
\fi%
\fi.}
\newcommand{\journal}[3]{#1, ``#2,'' in #3.} % pdfmatch fails when the #3 contains "{" or "}"
\newcommand{\book}[2]{#1,
 {\it #2}, in preparation.}
\newcommand{\patent}[3]{#1: ``#2,'' #3.} % patent number, title, date


% parameters:
% #1: name
% #2: title
% #3: organization
% #4: phone number
% #5: email
% #6: relationship
\newcommand{\reference}[6]{%
\begin{tabularx}{18cm}{L{10cm}L{7cm}}
{\bf #1} & \multirow[t]{5}{7cm}{\it Relationship: #6} \\
#2 & \\ % #2: title
#3 & \\ % #3: organization
#4 & \\ % #4: phone number
#5 &  % #5: email
\end{tabularx}\vspace*{0.5cm}%
}

\makeatletter
\def\input@path{{sections/}}
\makeatother
