\usepackage[a4paper,margin=1.5cm]{geometry}
\usepackage[T1]{fontenc}
\usepackage{tabularx}
\setlength\parindent{0pt}

\newcolumntype{L}[1]{@{\hspace*{0.5cm}}
                     >{\raggedright\arraybackslash}p{#1}}
\newcolumntype{C}[1]{>{\centering  \arraybackslash}p{#1}}
\newcolumntype{R}[1]{>{\raggedleft \arraybackslash}p{#1}
                     @{\hspace*{0.1cm}}}
\arrayrulewidth=1.2pt
\pagenumbering{gobble}

\renewcommand{\title}[3]{
\begin{center}
\huge{\textbf{\textsc{#1}}}
\\[0.1in]
\end{center}

\begin{minipage}[t]{0.3\textwidth}
#2
\end{minipage}
\hfill
\begin{minipage}[t]{0.3\textwidth}
\flushright{
\vfill
#3
}
\end{minipage}
\\[0.1in]
}



% customise the \section command 
\renewcommand{\section}[1]{
\begin{tabularx}{18cm}{L{13cm}R{4cm}}
\hspace*{-0.5cm}\Large{\textbf{\textsc{#1}}} & \\
\hline
\end{tabularx}\\[-1pt] }

\renewcommand{\subsection}[3]{
\begin{tabularx}{18cm}{L{13cm}R{4cm}}
{\bf #1}#2 & #3
\end{tabularx}\\ }

\newcommand{\content}[1]{
\begin{tabularx}{18cm}{@{\hspace*{0.5cm}}p{17.5cm}}
#1
\end{tabularx}}

\newcommand{\bolditem}[2]{{\bf #1} #2 \\ }
\newcommand{\position}[1]{{\it #1}\\ }
\newcommand{\bulletitem}[1]{$\bullet$\hspace*{0.2cm} #1\\ }
\newcommand{\paper}[3]{#1 \\
``#2,''
in #3. \\}
